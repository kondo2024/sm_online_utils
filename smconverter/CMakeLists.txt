include_directories(include)

file(GLOB LIB_SOURCES "src/*.cc")
add_library(smconverter SHARED ${LIB_SOURCES})

target_link_libraries(smconverter
    ${ROOT_LIBRARIES}
    XMLParser
    -L${ANAROOT_LIB_DIR} anacore anasamurai anacatana
)

file(GLOB APP_SOURCES "apps/*.cc")

foreach(source_file ${APP_SOURCES})
    get_filename_component(exe_name ${source_file} NAME_WE)
    add_executable(${exe_name} ${source_file})
    target_link_libraries(${exe_name} smconverter)
    install(TARGETS ${exe_name} DESTINATION bin)
endforeach()

install(TARGETS smconverter DESTINATION lib)
install(DIRECTORY include/ DESTINATION include/smconverter)

set(SETUP_CONTENT
"#!/bin/bash
# Generated by CMake - Fixed Path Version

export PATH=\"${CMAKE_INSTALL_PREFIX}/${CMAKE_INSTALL_BINDIR}:\$PATH\"
export LD_LIBRARY_PATH=\"${CMAKE_INSTALL_PREFIX}/${CMAKE_INSTALL_LIBDIR}:\$LD_LIBRARY_PATH\"

echo \"[sm_online_utils] Environment set to: ${CMAKE_INSTALL_PREFIX}\"
")

file(WRITE "${CMAKE_CURRENT_BINARY_DIR}/setup_smconverter.sh" "${SETUP_CONTENT}")

install(FILES "${CMAKE_CURRENT_BINARY_DIR}/setup_smconverter.sh"
        DESTINATION bin
        PERMISSIONS OWNER_READ OWNER_WRITE OWNER_EXECUTE GROUP_READ GROUP_EXECUTE WORLD_READ WORLD_EXECUTE)
